## 200 岛屿数量
连通分量个数
```cpp
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        if (grid.empty()) {
            return 0;
        }
        
        int ans = 0;
        const int m = grid.size(), n = grid[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == '1') {
                    ++ans;
                    dfs(i, j, grid);
                }
            }
        }
        return ans;
    }
private:
    void dfs(int x, int y, vector<vector<char>>& grid) {
        if (x < 0 || y < 0 || x >= grid.size() || y >= grid[0].size() || grid[x][y] == '0') {
            return;
        }
        grid[x][y] = '0';
        dfs(x + 1, y, grid);
        dfs(x - 1, y, grid);
        dfs(x, y + 1, grid);
        dfs(x, y - 1, grid);
    }
};
```
## 463 岛屿的周长
```cpp
class Solution {
public:
    int islandPerimeter(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 1) {
                    return dfs(grid, i, j);
                }
            }
        }
        return 0;
    }

    int dfs(vector<vector<int>>& grid, int i, int j) {
        // 与1相邻的边界周长加1
        if (i < 0 || j < 0 || i >= grid.size() || j >= grid[0].size()) {
            return 1;
        }
        // 与1相邻的水周长加1
        if (grid[i][j] == 0) return 1;
        if (grid[i][j] == 2) return 0;
        grid[i][j] = 2;
        return dfs(grid, i + 1, j) + dfs(grid, i - 1, j)
                + dfs(grid, i, j + 1) + dfs(grid, i, j - 1);
    }
};
```

## 695 岛屿的最大面积
最大连通分量
```cpp
class Solution {
public:
    int maxAreaOfIsland(vector<vector<int>>& grid) {
        if(grid.empty()) return 0;
        
        int area = 0;
        int row = grid.size(), col = grid[0].size();
        for(int i = 0;i < row;i++)
            for(int j = 0;j < col;j++) {
                if(grid[i][j] == 1) {
                    area = max(area, dfs(grid, row, col, i, j));
                }
            }
        return area;
    }
private:
    int dfs(vector<vector<int>>& grid, int row, int col, int x, int y) {
        if(x < 0 || x >= row || y < 0 || y >= col || grid[x][y] == 0) return 0;
        
        grid[x][y] = 0;
        return 1 + dfs(grid, row, col, x, y - 1) + dfs(grid, row, col, x - 1, y)
                 + dfs(grid, row, col, x, y + 1) + dfs(grid, row, col, x + 1, y);
    }
};
```
