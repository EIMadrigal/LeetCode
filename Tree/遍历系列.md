## 94 二叉树中序遍历
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
 
// Recursive
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> ans;
        inorderTraversal(root, ans);
        return ans;
    }
private:
    void inorderTraversal(TreeNode* root, vector<int>& ans) {
        if(!root) {
            return;
        }
        
        inorderTraversal(root->left, ans);
        ans.push_back(root->val);
        inorderTraversal(root->right, ans);
    }
};


// Iterative
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        if(!root) {
            return {};
        }
        
        stack<TreeNode*> s;
        vector<int> ans;
        TreeNode* cur = root;
        while(cur || !s.empty()) {
            if(cur) {
                s.push(cur);
                cur = cur->left;
            }
            else {
                cur = s.top();
                s.pop();
                ans.push_back(cur->val);
                // 借助栈顶元素走向右子树
                cur = cur->right;
            }
        }
        
        return ans;
    }
};
```

## 103 二叉树的锯齿形层序遍历
```cpp
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        if (!root) {
            return {};
        }
        vector<vector<int>> result;
        bool isOdd = true;
        queue<TreeNode*> q;
        q.push(root);
        
        while (!q.empty()) {
            int n = q.size();
            vector<int> elem(n);
            for (int i = 0; i < n; i++) {
                TreeNode* now = q.front(); q.pop();
                if (now->left)
                    q.push(now->left);
                if (now->right)
                    q.push(now->right);
                int index = isOdd ? i : n - i - 1;
                elem[index] = now->val;
            }
            isOdd = !isOdd;
            result.push_back(elem);
        }
        return result;
    }
};
```
