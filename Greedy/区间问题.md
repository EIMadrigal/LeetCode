考虑按照区间**开始**或**结尾**排序
## 435 无重叠区间
按照区间结尾排序，结尾越小留给其他区间的空间就越大，就可以保留更多区间，因此优先保留结尾小且和前一个选中区间不相交的区间
```cpp
class Solution {
public:
    int eraseOverlapIntervals(vector<vector<int>>& intervals) {
        const int n = intervals.size();
        sort(intervals.begin(), intervals.end(), [](vector<int>& a, vector<int>& b) {
            return a[1] < b[1];
        });
        int res = 0, last = 0;
        for (int i = 1; i < n; ++i) {
            int s1 = intervals[last][0], e1 = intervals[last][1];
            int s2 = intervals[i][0], e2 = intervals[i][1];
            if (s1 < e2 && s2 < e1) {  // 区间相交
                ++res;
            } else {
                last = i;
            }
        }
        return res;
    }
};
```

## 452 用最少数量的箭引爆气球
```cpp
class Solution {
public:
    int findMinArrowShots(vector<vector<int>>& points) {
        const int n = points.size();
        sort(points.begin(), points.end(), [](vector<int>& a, vector<int>& b) {
            return a[1] < b[1];
        });
        int res = 1, last = 0;
        for (int i = 1; i < n; ++i) {
            int s1 = points[last][0], e1 = points[last][1];
            int s2 = points[i][0], e2 = points[i][1];
            if (s1 <= e2 && s2 <= e1) {  // 区间相交
                continue;
            } else {
                ++res;
                last = i;
            }
        }
        return res;
    }
};
```

## 763 划分字母区间
预处理：频率，第一次出现位置，最后一次出现位置
```cpp
class Solution {
public:
    vector<int> partitionLabels(string s) {
        vector<int> last(26, -1);
        for (int i = 0; i < s.size(); ++i) {
            last[s[i] - 'a'] = i;
        }
        vector<int> res;
        int l = 0, r = 0;
        for (int i = 0; i < s.size(); ++i) {
            r = max(r, last[s[i] - 'a']);
            if (i == r) {
                res.emplace_back(r - l + 1);
                l = r + 1;
            }
        }
        return res;
    }
};
```

## 646 最长数对链
```cpp
class Solution {
public:
    int findLongestChain(vector<vector<int>>& pairs) {
        const int n = pairs.size();
        sort(pairs.begin(), pairs.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[1] < b[1] || (a[1] == b[1] && a[0] < b[0]);
        });
        int res = 1, last = pairs[0][1];
        for (int i = 1; i < n; ++i) {
            if (pairs[i][0] > last) {
                ++res;
                last = pairs[i][1];
            }
        }
        return res;
    }
};
```

## 1235 规划兼职工作
```cpp
class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        const int n = startTime.size();
        vector<vector<int>> nums(n);
        for (int i = 0; i < n; ++i) {
            nums[i] = {startTime[i], endTime[i], profit[i]};
        }
        sort(nums.begin(), nums.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[1] < b[1];      
        });
        vector<int> dp(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            // 在endTime中二分i的startTime, 因此endTime有序
            int l = 1, r = i + 1;
            while (l < r) {
                int m = l + (r - l) / 2;
                if (nums[i - 1][0] >= nums[m - 1][1]) {
                    l = m + 1;
                } else {
                    r = m;
                }
            }
            dp[i] = max(dp[i - 1], dp[l - 1] + nums[i - 1][2]);
        }
        return dp[n];
    }
};
```
