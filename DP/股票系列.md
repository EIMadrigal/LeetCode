## 121 买卖股票的最佳时机
只能买卖一次
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        const int n = prices.size();
        if (0 == n)
            return 0;
        int curMin = prices[0], curProf = 0;
        for (int i = 1; i < n; ++i) {
            if (prices[i] < curMin)
                curMin = prices[i];
            curProf = max(curProf, prices[i] - curMin);
        }
        return curProf;
    }
};
```
