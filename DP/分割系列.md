## 279 完全平方数
```cpp
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1);
        for (int i = 1; i <= n; i++) {
            int cnt = INT_MAX;
            for (int j = 1; j * j <= i; j++) {
                cnt = min(cnt, dp[i - j * j]);
            }
            dp[i] = cnt + 1;
        }
        return dp[n];
    }
};
```

## 91 解码方法
```cpp
class Solution {
public:
    int numDecodings(string s) {
        if (s[0] == '0') return 0;
        int n = s.length();
        vector<int> dp(n + 1, 0);
        dp[0] = 1, dp[1] = 1;
        for (int i = 2; i <= n; ++i) {
            if (s[i - 1] != '0')
                dp[i] = dp[i - 1];
            if (s[i - 2] == '1' || (s[i - 2] == '2' && s[i - 1] <= '6')) {
                dp[i] += dp[i - 2];
            }
        }
        return dp[n];
    }
};
```

## 139 单词拆分
```cpp
class Solution {
public:
	bool wordBreak(string s, vector<string>& wordDict) {
		unordered_set<string> words(wordDict.cbegin(), wordDict.cend());
		const int n = s.size();
		vector<bool> dp(n + 1, false);
		dp[0] = true;
		for (int i = 1; i <= n; ++i) {
            for (int j = 0; j < i; ++j) {
				if (dp[j] && words.find(s.substr(j, i - j)) != words.end()) {
					dp[i] = true;
					break;
				}
			}
        }
		return dp[n];
	}
};
```
