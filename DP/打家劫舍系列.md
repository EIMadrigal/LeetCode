class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        if (n < 2) return nums[0];
        vector<int> dp(n, 0);
        dp[0] = nums[0], dp[1] = max(nums[0], nums[1]);
        for (int i = 2; i < n; ++i) {
            dp[i] = max(dp[i - 2] + nums[i], dp[i - 1]);
        }
        return dp[n - 1];
    }
};

class Solution {
public:
    int rob(vector<int>& nums) {
        int rob1 = 0, rob2 = 0, cur = 0;
        for (int a : nums) {
            cur = max(rob1, rob2 + a);
            rob2 = rob1;
            rob1 = cur;
        }
        return cur;
    }
};
