## 496 Next Greater Element
```python
class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        stack = []
        ans = {}
        for num in nums2:
            while stack and num > stack[-1]:
                cur = stack.pop()
                ans[cur] = num
            stack.append(num)
        while stack:
            cur = stack.pop()
            ans[cur] = -1
        return [ans[num] for num in nums1]
```
## 503 Next Greater Element II
```python
# 对循环数组，取余访问2倍数组即可
class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        ans = [-1] * len(nums)
        stack = []
        for i in range(len(nums) * 2 - 1):
            while stack and nums[i % len(nums)] > nums[stack[-1]]:
                cur = stack.pop()
                ans[cur] = nums[i % len(nums)]
            stack.append(i % len(nums))
        return ans
```
## 739 Daily Temperatures
```python
# 单调栈自底向上由大到小
class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        ans = [0] * len(temperatures)
        stack = []
        for i, t in enumerate(temperatures):
            while stack and t > temperatures[stack[-1]]:
                cur_idx = stack.pop()
                ans[cur_idx] = i - cur_idx
            stack.append(i)
        return ans
```

## 1856 子数组最小乘积的最大值
```cpp
class Solution {
public:
    int maxSumMinProduct(vector<int>& nums) {
        const int n = nums.size(), mod = 1e9 + 7;
        vector<long long> preSum(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            preSum[i] = preSum[i - 1] + nums[i - 1];    
        }
        long long res = 0;
        stack<int> s;
        for (int i = 0; i < n; ++i) {
            while (!s.empty() && nums[i] <= nums[s.top()]) { // 右侧非严格小于
                int idx = s.top(); s.pop();
                int l = s.empty() ? -1 : s.top();
                int r = i;
                long long partialSum = preSum[r] - preSum[l + 1];
                res = max(res, partialSum * nums[idx]);
            }
            s.push(i);
        }
        while (!s.empty()) {
            int idx = s.top(); s.pop();
            int l = s.empty() ? -1 : s.top();
            int r = n;
            long long partialSum = preSum[r] - preSum[l + 1];
            res = max(res, partialSum * nums[idx]);
        }
        return res % mod;
    }
};
```
