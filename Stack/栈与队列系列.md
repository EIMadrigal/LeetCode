## 232 用栈实现队列
```cpp
class MyQueue {
public:
    /** Initialize your data structure here. */
    MyQueue() {
        
    }
    
    /** Push element x to the back of queue. */
    void push(int x) {
        end.push(x);
    }
    
    /** Removes the element from in front of queue and returns that element. */
    int pop() {
        if (front.empty()) {
            while (!end.empty()) {
                front.push(end.top());
                end.pop();
            }
        }
        int ans = front.top();
        front.pop();
        return ans;
    }
    
    /** Get the front element. */
    int peek() {
        if (front.empty()) {
            while (!end.empty()) {
                front.push(end.top());
                end.pop();
            }
        }
        return front.top();
    }
    
    /** Returns whether the queue is empty. */
    bool empty() {
        return front.empty() && end.empty();
    }
private:
    stack<int> front;
    stack<int> end;
};
```
