## 148 排序链表
快排
```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        ListNode preHead(-1, head), *dummy = &preHead;
        quickSort(dummy, nullptr);
        return dummy->next;
    }

    void quickSort(ListNode* head, ListNode* tail) {
        if (head == tail || head->next == tail) {
            return;
        }
        ListNode* pivot = head->next, *prev = pivot;
        for (ListNode* cur = pivot->next; cur != tail; ) {
            if (cur->val < pivot->val) {
                prev->next = cur->next;
                cur->next = head->next;
                head->next = cur;
                cur = prev->next;
            }
            else {
                prev = cur;
                cur = cur->next;
            }
        }
        quickSort(head, pivot);
        quickSort(pivot, tail);
    }
};
```

top-down归并
```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        return mergeSort(head, nullptr);
    }

    ListNode* mergeSort(ListNode* head, ListNode* tail) {
        if (head == nullptr) {
            return head;
        }
        if (head->next == tail) {
            head->next = nullptr;
            return head;
        }
        ListNode* fast = head, *slow = head;
        while (fast != tail && fast->next != tail) {
            fast = fast->next->next;
            slow = slow->next;
        }
        ListNode* l = mergeSort(head, slow);
        ListNode* r = mergeSort(slow, tail);
        return merge(l, r);
    }

    ListNode* merge(ListNode* l, ListNode* r) {
        ListNode preHead(-1), *cur = &preHead;
        ListNode* tmp1 = l, *tmp2 = r;
        while (tmp1 && tmp2) {
            if (tmp1->val < tmp2->val) {
                cur->next = tmp1;
                tmp1 = tmp1->next;
            }
            else {
                cur->next = tmp2;
                tmp2 = tmp2->next;
            }
            cur = cur->next;
        }
        if (tmp1) {
            cur->next = tmp1;
        }
        else if (tmp2) {
            cur->next = tmp2;
        }
        return preHead.next;
    }
};
```
