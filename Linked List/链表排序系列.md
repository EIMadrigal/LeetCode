## 148 排序链表
快排
```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        ListNode preHead(-1, head), *dummy = &preHead;
        quickSort(dummy, nullptr);
        return dummy->next;
    }

    void quickSort(ListNode* head, ListNode* tail) {
        if (head == tail || head->next == tail) {
            return;
        }
        ListNode* pivot = head->next, *prev = pivot;
        for (ListNode* cur = pivot->next; cur != tail; ) {
            if (cur->val < pivot->val) {
                prev->next = cur->next;
                cur->next = head->next;
                head->next = cur;
                cur = prev->next;
            }
            else {
                prev = cur;
                cur = cur->next;
            }
        }
        quickSort(head, pivot);
        quickSort(pivot, tail);
    }
};
```

top-down归并
```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        return mergeSort(head, nullptr);
    }

    ListNode* mergeSort(ListNode* head, ListNode* tail) {
        if (head == nullptr) {
            return head;
        }
        if (head->next == tail) {
            head->next = nullptr;
            return head;
        }
        ListNode* fast = head, *slow = head;
        while (fast != tail && fast->next != tail) {
            fast = fast->next->next;
            slow = slow->next;
        }
        ListNode* l = mergeSort(head, slow);
        ListNode* r = mergeSort(slow, tail);
        return merge(l, r);
    }

    ListNode* merge(ListNode* l, ListNode* r) {
        ListNode preHead(-1), *cur = &preHead;
        ListNode* tmp1 = l, *tmp2 = r;
        while (tmp1 && tmp2) {
            if (tmp1->val < tmp2->val) {
                cur->next = tmp1;
                tmp1 = tmp1->next;
            }
            else {
                cur->next = tmp2;
                tmp2 = tmp2->next;
            }
            cur = cur->next;
        }
        if (tmp1) {
            cur->next = tmp1;
        }
        else if (tmp2) {
            cur->next = tmp2;
        }
        return preHead.next;
    }
};
```

bottom-up归并
```cpp
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) {
            return head;
        }
        int length = 0;
        ListNode* tmp = head;
        while (tmp) {
            tmp = tmp->next;
            ++length;
        }
        ListNode preHead(-1, head), *dummy = &preHead;
        for (int len = 1; len < length; len <<= 1) {
            ListNode* cur = dummy->next, *prev = dummy;
            while (cur) {
                ListNode* head1 = cur;
                for (int i = 1; i < len && cur->next; ++i) {
                    cur = cur->next;
                }
                ListNode* head2 = cur->next;
                cur->next = nullptr;
                cur = head2;

                for (int i = 1; i < len && cur; ++i) {
                    cur = cur->next;
                }
                ListNode* nxt = nullptr;
                if (cur) {
                    nxt = cur->next;
                    cur->next = nullptr;
                }
                prev->next = merge(head1, head2);
                while (prev->next) {
                    prev = prev->next;
                }
                cur = nxt;
            }
        }
        return dummy->next;
    }

    ListNode* merge(ListNode* head1, ListNode* head2) {
        ListNode* dummyHead = new ListNode(0);
        ListNode* temp = dummyHead, *temp1 = head1, *temp2 = head2;
        while (temp1 != nullptr && temp2 != nullptr) {
            if (temp1->val <= temp2->val) {
                temp->next = temp1;
                temp1 = temp1->next;
            } else {
                temp->next = temp2;
                temp2 = temp2->next;
            }
            temp = temp->next;
        }
        if (temp1 != nullptr) {
            temp->next = temp1;
        } else if (temp2 != nullptr) {
            temp->next = temp2;
        }
        return dummyHead->next;
    }
};
```
